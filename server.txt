add_header X-Frame-Options "SAMEORIGIN";
add_header X-Content-Type-Options "nosniff";
add_header X-XSS-Protection "1; mode=block";

server {
    listen 80;
    listen [::]:80;
    server_name aisteel.unimed.ac.id;
    return 301 https://aisteel.unimed.ac.id;
}

server {
    listen 443 ssl;
    listen [::]:443 ssl;
    server_name aisteel.unimed.ac.id;
    client_body_timeout 5s;
    client_header_timeout 5s;

    error_log /var/log/nginx/aisteel.unimed.ac.id.error.log warn;

    include /etc/nginx/ssl.d/ssl.conf;
    include /etc/nginx/security.d/security.conf;

    root /var/www/webapp/public;

    index index.php;

    charset utf-8;

    location / {
        try_files $uri $uri/ /index.php?$query_string;
    }

    location = /favicon.ico { access_log off; log_not_found off; }
    location = /robots.txt  { access_log off; log_not_found off; }

    error_page 404 /index.php;

    location ~ ^/index\.php(/|$) {
        fastcgi_pass unix:/var/run/php/php8.2-fpm.sock;
        fastcgi_param SCRIPT_FILENAME $realpath_root$fastcgi_script_name;
        include fastcgi_params;
        fastcgi_hide_header X-Powered-By;
    }

    location ~ \.php(.*)$ {
        include /etc/nginx/fastcgi.d/fastcgi.conf;
    }

    location ~ /\.(?!well-known).* {
        deny all;
    }
}
